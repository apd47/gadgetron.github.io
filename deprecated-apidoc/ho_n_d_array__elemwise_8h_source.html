89  // Utility to verify if array dimensions are compatible for a binary operation
90  // that supports simple broadcasting, i.e. for f(x,y,r), there are three cases:
91  // 1) nr = nx = ny
92  // 2) nr = nx > ny, and nx is divisible by ny
93  // 3) nr = ny > nx, and ny is divisible by nx
94  //
95  template<class T,class S, class U> bool compatible_dimensions( const hoNDArray<T> &x, const hoNDArray<S> &y,
96  const hoNDArray<U> &r )
97  {
98  size_t nx = x.get_number_of_elements();
99  size_t ny = y.get_number_of_elements();
100  size_t nr = r.get_number_of_elements();
101  if (nx == ny) {
102  return (nx==nr);
103  }
104  if ((nx%ny)==0) {
105  return (nx==nr);
106  }
107  if ((ny%nx)==0) {
108  return (ny==nr);
109  }
110  return false;
111  }
