453  * @brief Gold section search algorithm. Only works with unimodal functions, which we assume we're dealing with, at least locally
454  * @param f Functor to calculate the function to minimize
455  * @param a Start of the bracketing
456  * @param d End of bracketing
457  * @return Value minimizing the function f.
458  */
459  REAL gold(FunctionEstimator& f, REAL a, REAL d){
460  const REAL gold = 1.0/(1.0+std::sqrt(5.0))/2;
461 
462  REAL b = d-(d-a)*gold;
463  REAL c = (d-a)*gold-a;
464 
465  REAL fa = f(a);
466  REAL fb = f(b);
467  REAL fc = f(c);
468  REAL fd = f(d);
469  REAL tol = 1e-6;
470 
471  while (abs(a-d) > tol*(abs(b)+abs(c))){
472  if (fb > fc){
473  a = b;
474  fa = fb;
475  b = c;
476  fb = fc;
477  c= b*gold+(1.0-gold)*d;
478  fc = f(c);
479  } else {
480  d = c;
481  fd = fc;
482  c = b;
483  fc = fb;
484  b = c*gold+(1-gold)*a;
485  fb = f(b);
486  }
487  }
488  if (fb < fc){
489  f(b);
490  return b;
491  }else {
492  f(c);
493  return c;
494  }

617  REAL cg_linesearch(FunctionEstimator& f, const REAL alpha0, const REAL gd, const REAL old_norm){
618  REAL delta=0.1;
619  REAL sigma=0.9;
620  REAL nabla=0.66;
621  //REAL precision = 0.0003; //Estimated precision of function evaluation
622  REAL precision = 1e-4f; //Estimated precision of function evaluation
623 
624 
625 
626 
627  REAL a=0;
628  REAL b = alpha0;
629 
630  REAL ak = a;
631  REAL bk = b;
632  REAL fa = old_norm;
633  ELEMENT_TYPE a_deriv = gd;
634  REAL fb = f(alpha0);
635  ELEMENT_TYPE b_deriv = f.dir_deriv();
636 
637  while (abs(a-b) > 0){
638  if ((((2*delta-1.0)*real(gd) >= real(b_deriv)) && (fb < old_norm+precision)) && //Check Approximate Wolfe conditions
639  (abs(b_deriv) <= sigma*abs(gd))){
640  f(b);
641  return b;
642  }
643 
644  if ((((2*delta-1.0)*real(gd) >= real(a_deriv)) && (fa < old_norm+precision)) && //Check Approximate Wolfe conditions
645  (abs(a_deriv) <= sigma*abs(gd))){
646  f(a);
647  return a;
648  }
649 
650  secant2(a,b,f,old_norm+precision);
651  if ((b-a) > nabla*(bk-ak)) {
652  REAL c = (a+b)/2;
653  interval_update(a,b,c,f,old_norm);
654  }
655  if (a != ak){
656  fa = f(a);
657  a_deriv = f.dir_deriv();
658  }
659
