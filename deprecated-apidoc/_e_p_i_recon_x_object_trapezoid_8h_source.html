83  trajectoryNeg_.create(numSamples_);

95 
96  // Some timings
97  float totTime = rampUpTime_ + flatTopTime_ + rampDownTime_;
98  float readTime = dwellTime_ * numSamples_;
99 
100  // Fix the acqDelayTime for balanced acquisitions
101  if (balanced_) {
102  acqDelayTime_ = 0.5 * (totTime - readTime);
103  }
104 
105  // Some Areas
106  float totArea = 0.5*rampUpTime_ + flatTopTime_ + 0.5*rampDownTime_;
107  float readArea = 0.5*rampUpTime_ + flatTopTime_ + 0.5*rampDownTime_;
108  if (rampUpTime_ > 0.0) {
109  readArea -= 0.5*(acqDelayTime_)*(acqDelayTime_)/rampUpTime_;
110  }
111  if (rampDownTime_ > 0.0) {
112  readArea -= 0.5*(totTime - (acqDelayTime_+readTime))*(totTime - (acqDelayTime_+readTime))/rampDownTime_;
113  }
114 
115  // Prephase is set so that k=0 is halfway through the readout time
116  float prePhaseArea = 0.5 * totArea;
117 
118  // The scale is set so that the read out area corresponds to the number of encoded points
119  float scale = encodeNx_ /readArea;
120 
121  for (n=0; n<nK; n++)
122  {
123  t = (n+1.0)*dwellTime_ + acqDelayTime_; // end of the dwell time
124  if (t <= rampUpTime_) {
125  // on the ramp up
126  k[n] = 0.5 / rampUpTime_ * t*t;
127  }
128  else if ((t > rampUpTime_) && (t <= (rampUpTime_+flatTopTime_))) {
129  // on the flat top
130  k[n] = 0.5*rampUpTime_ + (t - rampUpTime_);
131  }
132  else {
133  // on the ramp down
134  float v = (rampUpTime_+flatTopTime_+rampDownTime_-t);
135  k[n] = 0.5*rampUpTime_ + flatTopTime_ + 0.5*rampDownTime_ - 0.5/rampDownTime_*v*v;
136  }
137  //GDEBUG_STREAM(n << ": " << t << " " << k[n] << " " << std::endl);
138  }
139 
140  // Fill the positive and negative trajectories
141  for (n=0; n<numSamples_; n++)
142  {
143  trajectoryPos_[n] = scale * (k[n] - prePhaseArea);
144  trajectoryNeg_[n] = scale * (-1.0*k[n] + totArea - prePhaseArea);
145  //GDEBUG_STREAM(n << ": " << trajectoryPos_[n] << " " << trajectoryNeg_[n] << std::endl);
146  }
147

149  operatorComputed_ = false;
