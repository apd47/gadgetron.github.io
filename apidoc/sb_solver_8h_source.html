191  virtual void update_dk(ARRAY_TYPE_ELEMENT* u_k)
192  {
193  ARRAY_TYPE_REAL s_k(codom_dims);
194  ARRAY_TYPE_ELEMENT *tmp = new ARRAY_TYPE_ELEMENT[reg_ops.size()];
195  for (int i=0; i<reg_ops.size(); i++) {
196  tmp[i] = *b_ks[i];
197  this->reg_ops[i]->mult_M(u_k,&tmp[i],true);
198  if (this->prior.get())
199  tmp[i] -= *p_Ms[i];
200  (i==0) ? s_k = *abs_square<ELEMENT_TYPE>(&tmp[i]) : s_k += *abs_square<ELEMENT_TYPE>(&tmp[i]);
201  }
202  sqrt_inplace(&s_k);
203  for (int i=0; i<reg_ops.size(); i++) {
204  shrinkd(&tmp[i],&s_k,REAL(1)/reg_ops[i]->get_weight(),d_ks[i].get());
205  }
206  delete[] tmp;
207  }

557  //

581

583  //

701  virtual void core( REAL tolerance, unsigned int outer_iterations, unsigned int inner_iterations,
702  boost::shared_ptr<ARRAY_TYPE_ELEMENT> f,
703  boost::shared_ptr<ARRAY_TYPE_ELEMENT> u_k )
704  {
705  // Image space dimensions
706  boost::shared_ptr< std::vector<size_t> > image_dims =
707  this->encoding_operator_->get_domain_dimensions();
708 
709  // Keep a copy of the "previous" u_k to compute the outer loop change of u_k
710  //
711 
712  ARRAY_TYPE_ELEMENT u_k_prev;
713  if( tolerance > REAL(0) || this->output_mode_ >= solver<ARRAY_TYPE_ELEMENT, ARRAY_TYPE_ELEMENT>::OUTPUT_VERBOSE ){
714  u_k_prev = *u_k;
715  }
716 
717  //
718  // Outer loop
719  //
720 
721  for( unsigned int outer_iteration=0; outer_iteration<outer_iterations; outer_iteration++ ) {
722 
723  if( this->output_mode_ >= solver<ARRAY_TYPE_ELEMENT, ARRAY_TYPE_ELEMENT>::OUTPUT_MAX )
724  GDEBUG_STREAM(std::endl << "SB outer loop iteration " << outer_iteration << std::endl << std::endl);
725 
726  //
727  // Inner loop
728  //
729 
730  for( unsigned int inner_iteration=0; inner_iteration<inner_iterations; inner_iteration++ ) {
731 
732  if( this->output_mode_ >= solver<ARRAY_TYPE_ELEMENT, ARRAY_TYPE_ELEMENT>::OUTPUT_MAX )
733  GDEBUG_STREAM(std::endl << "SB inner loop iteration " << inner_iteration << std::endl << std::endl);
734 
735  { // Brackets used to free 'data' below as soon as it goes out of scope
736 
737  // Setup input vector to the encoding operator container (argument to the inner solver's solve)
738  //
739 
740  ARRAY_TYPE_ELEMENT data(enc_op_container_->get_codomain_dimensions());
741  ARRAY_TYPE_ELEMENT tmp(f->get_dimensions().get(), data.get_data_ptr() );
742 
743  tmp = *f;
744 
745  // Next add the regularization operators' data, d_k - b_k
746  //
747 
748  for( unsigned int i=0; i< regularization_operators_.size(); i++ ){
749  boost::shared_ptr<sbRegularizationOperator > op = regularization_operators_[i];
750  tmp.create( op->get_codomain_dimensions(), data.get_data_ptr()+enc_op_container_->get_offset(i+1) );
751  op->update_encoding_space(&tmp);
752  }
753 
754  // Solve for u_k
755  //
756 
757  {
758  if (use_x0_){
759  get_inner_solver()->set_x0(u_k);
760  }
761 
762  boost::shared_ptr<ARRAY_TYPE_ELEMENT> tmp_u_k =
763  get_inner_solver()->solve( &data );
764 
765  // Invoke the post inner solver callback
766  post_linear_solver_callback( tmp_u_k.get() );
767 
768  // Compute change in u_k
769  if( this->output_mode_ >= solver<ARRAY_TYPE_ELEMENT, ARRAY_TYPE_ELEMENT>::OUTPUT_VERBOSE ){
770  *u_k -= *tmp_u_k;
771  GDEBUG_STREAM("u_k delta l2-norm (inner loop): " << nrm2(u_k.get()) << std::endl);
772  }
773 
774  // Update u_k
775  *u_k = *tmp_u_k;
776  }
777  }
778 
779  // Update d_k (and b_k in final inner iteration)
780  //
781 
782  for( unsigned int i=0; i< regularization_operators_.size(); i++ ){
783  boost::shared_ptr<sbRegularizationOperator > op = regularization_operators_[i];
784  if( inner_iteration < inner_iterations-1 )
785  op->update_dk(u_k.get());
786  else
787  op->update_dk_bk(u_k.get());
788  }
789  } // end of inner loop
790 
791  // Output change in u_k
792  if( tolerance > REAL(0) || this->output_mode_ >= solver<ARRAY_TYPE_ELEMENT, ARRAY_TYPE_ELEMENT>::OUTPUT_VERBOSE ){
793  u_k_prev *= ELEMENT_TYPE(-1);
794  u_k_prev += *u_k;
795  REAL delta = nrm2(&u_k_prev);
796 
797  if( this->output_mode_ >= solver<ARRAY_TYPE_ELEMENT, ARRAY_TYPE_ELEMENT>::OUTPUT_VERBOSE )
798  GDEBUG_STREAM("u_k delta l2-norm (outer loop): " << delta << std::endl << std::endl);
799 
800  if( delta < tolerance )
801  break;
802 
803  u_k_prev = *u_k;
804  }
805  } // end of outer loop
806  }

816  //

819  ARRAY_TYPE_ELEMENT tmp( op->get_domain_dimensions() );
820  op->mult_MH( f, &tmp );

822  image_scale = REAL(tmp.get_number_of_elements())/sum;
823  *f *= image_scale;
