270  acquisition_->get_geometry()->get_offsets(),
271  binning_->get_bin(b),
272  projections_per_batch_, is_dims_lores, is_dims_in_mm_, ps_dims_in_mm,
273  SDD, SAD, false, use_offset_correction_, false );
