346  bool hoImageRegDeformationFieldRegister<ValueType, CoordType, D>::performRegistration()
347  {
348  try
349  {
350  // starting from the most coarse level
351 
352  int level;
353  for ( level=(int)resolution_pyramid_levels_-1; level>=0; level-- )
354  {
355  // update the transform for multi-resolution pyramid
356  transform_->update();
357 
358  // GADGET_CHECK_RETURN_FALSE(solver_pyramid_[level].initialize());
359  GADGET_CHECK_RETURN_FALSE(solver_pyramid_[level].solve());
360 
361  if ( !debugFolder_.empty() )
362  {
363  unsigned int jj;
364  for ( jj=0; jj<D; jj++ )
365  {
366  std::ostringstream ostr;
367  ostr << "deform_" << jj;
368 
369  gt_exporter_.exportImage(transform_->getDeformationField(jj), debugFolder_+ostr.str());
370  }
371  }
372 
373  // expand the deformation field for next resolution level
374  if ( level>0 )
375  {
376  std::vector<float> ratio = resolution_pyramid_downsample_ratio_[level-1];
377 
378  unsigned int jj;
379  bool downsampledBy2 = true;
380  for ( jj=0; jj<D; jj++ )
381  {
382  if ( std::abs(ratio[jj]-2.0f) > FLT_EPSILON )
383  {
384  downsampledBy2 = false;
385  break;
386  }
387  }
388 
389  DeformationFieldType deformExpanded;
390  deformExpanded.createFrom(target_pyramid_[level-1]);
391  // Gadgetron::clear(deformExpanded);
392  memset(deformExpanded.begin(), 0, deformExpanded.get_number_of_bytes());
393 
394  if ( downsampledBy2 || resolution_pyramid_divided_by_2_ )
395  {
396  for ( jj=0; jj<D; jj++ )
397  {
398  DeformationFieldType& deform = transform_->getDeformationField(jj);
399  Gadgetron::expandImageBy2(deform, *target_bh_pyramid_construction_, deformExpanded);
400 
401  if ( !use_world_coordinates_ )
402  {
403  Gadgetron::scal(ValueType(2.0), deformExpanded); // the deformation vector should be doubled in length
404  }
405 
406  deform = deformExpanded;
407  }
408  }
409  else
410  {
411  for ( jj=0; jj<D; jj++ )
412  {
413  DeformationFieldType& deform = transform_->getDeformationField(jj);
414  Gadgetron::upsampleImage(deform, *target_interp_pyramid_construction_, deformExpanded, &ratio[0]);
415 
416  if ( !use_world_coordinates_ )
417  {
418  Gadgetron::scal(ValueType(ratio[jj]), deformExpanded);
419  }
420 
421  deform = deformExpanded;
422  }
423  }
424 
425  if ( !debugFolder_.empty() )
426  {
427  for ( jj=0; jj<D; jj++ )
428  {
429  std::ostringstream ostr;
430  ostr << "deformExpanded_" << jj;
431 
432  gt_exporter_.exportImage(transform_->getDeformationField(jj), debugFolder_+ostr.str());
433  }
434  }
435  }
436  }
437  }

439  {
